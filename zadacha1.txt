movements!(r::Robot, side::HorizonSide, num_steps::Int) = for _ in 1:num_steps move!(r,side) end

function mark_kross(r)
    side=Nord
    for i in 0:1:3
    steps=0
    while isborder(r,side)==false
    move!(r,side)
    putmarker!(r)
    steps+=1
    end
    movements!(r,inverse(side),steps)
    side=HorizonSide(mod(Int(side)+1, 4))
    end
    putmarker!(r)
    end
